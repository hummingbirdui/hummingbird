### Initialization

This example shows how to initialize **Hummingbird UI** components dynamically when the route changes in a React application using `react-router`. It uses `useEffect` to load the full library only on the client.

<Example>
  <div slot="code">
    ```js
    import { useEffect } from "react";
    import { useLocation } from "react-router";

    async function loadHummingbird() {
      return import("@hummingbirdui/hummingbird");
    }

    function App() {
      const location = useLocation();

      useEffect(() => {
        const initHummingbird = async () => {
          await loadHummingbird();
        };
        initHummingbird();
      }, [location.pathname]);

      return (
        ...
      );
    }

    export default App;
    ```

  </div>
</Example>

### JavaScript API

This example demonstrates how to use the **JavaScript API** for interacting with a Hummingbird component â€” in this case, a `Modal`. The modal instance is created and stored in local state, allowing it to be controlled programmatically using `toggle()`.

Type support for the instance (e.g., `ModalInstance`) ensures correct usage with TypeScript.

<Example>
  <div slot="code">
    ```js
    import { useEffect, useState } from "react";
    import { useLocation } from "react-router";
    import type { ModalInstance } from "@hummingbirdui/hummingbird";

    async function loadHummingbird() {
      return import("@hummingbirdui/hummingbird");
    }

    function Component() {
      const location = useLocation();
      const [myModal, setMyModal] = useState<ModalInstance>();

      useEffect(() => {
        const initModal = async () => {
          const { Modal } = await loadHummingbird();
          setMyModal(new Modal(".modal"));
        };
        initModal();
      }, [location.pathname]);

      const handleClick = () => {
        myModal?.toggle();
      };

      return (
        <div>
          <button className="btn btn-primary" onClick={handleClick}>Primary</button>
          <div className="modal">...</div>
        </div>
      );
    }

    export default Component;
    ```

  </div>
</Example>
